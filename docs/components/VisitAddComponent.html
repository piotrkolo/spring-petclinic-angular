import {Component, OnInit} from '@angular/core';
import {Visit} from '../note';
import {VisitService} from '../note.service';
import {ActivatedRoute, Router} from '@angular/router';
import {CaseService} from '../../cases/case.service';
import {Case} from '../../cases/case';
import {Status} from '../../statuses/status';
import {User} from '../../users/user';

import * as moment from 'moment';

@Component({
  selector: 'app-note-add',
  templateUrl: './note-add.component.html',
  styleUrls: ['./note-add.component.css']
})
export class VisitAddComponent implements OnInit {

  note: Visit;
  currentCase: Case;
  currentUser: User;
  currentStatus: Status;
  addedSuccess = false;
  errorMessage: string;

  constructor(private visitService: VisitService, private caseService: CaseService, private router: Router, private route: ActivatedRoute) {
    this.note = {} as Visit;
    this.currentCase = {} as Case;
    this.currentUser = {} as User;
    this.currentStatus = {} as Status;

  }

  ngOnInit() {
    console.log(this.route.parent);
    const caseId = this.route.snapshot.params.id;
    this.caseService.getCaseById(caseId).subscribe(
      response => {
        this.currentCase = response;
        this.note.case = this.currentCase;
        this.currentStatus = this.currentCase.type;
        this.currentUser = this.currentCase.user;
      },
      error => this.errorMessage = error as any);
  }

  onSubmit(note: Visit) {
    note.id = null;
    const that = this;

    // format output from datepicker to short string yyyy/mm/dd format
    note.date = moment(note.date).format('YYYY/MM/DD');


    this.visitService.addVisit(note).subscribe(
      newVisit => {
        this.note = newVisit;
        this.addedSuccess = true;
        that.gotoUserDetail();
      },
      error => this.errorMessage = error as any
    );
  }

  gotoUserDetail() {
    this.router.navigate(['/users', this.currentUser.id]);
  }

}
